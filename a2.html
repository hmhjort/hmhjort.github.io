Here is a code snipit from the Arduino code that I wrote for the circuit. I used red, yellow, and green LEDs and connected them to the 5, 6, and 9 pins in my Arduino. I also used a pushbutton which I connected to the 10 pin on my Arduino. I chose to have both the yellow and green LEDs each light up for 1000ms and then turn off for 1000ms. I chose to have the red LED slowly fade from 0 to 255 PWM pausing for 9ms between each inscrease in PWM. Then there is a 1000ms pause before the loop starts again. 


        // Label LED pins and button pin
        int red = 9; // Pin for red LED
        int yellow = 5; // Pin for yellow LED
        int green = 6; // Pin for green LED
        int buttonPin = 10; // Pin for button

        int buttonState = 0; // variable for reading buttonpin

        // the setup function runs once when you press reset or power the board
        void setup() {
          // Set all LED pins as OUTPUT
          pinMode(yellow, OUTPUT);  // initialize yellow LED as an output.
          pinMode(green, OUTPUT);   // initialize green LED as an output.
          pinMode(red, OUTPUT);   // initialize red LED as an output.

          pinMode(buttonPin, INPUT); // initialize the pushbutton pin as an input.
        }

        // the loop function runs over and over again forever
        void loop() {
          // turn red LED on and off 
          digitalWrite(yellow, HIGH);  // turn the yellow LED on (HIGH is the voltage level)
          delay(1000);                      // wait for 1000 milliseconds
          digitalWrite(yellow, LOW);   // turn the yellow LED off by making the voltage LOW
          delay(1000);      // wait for 1000 milliseconds
        
          digitalWrite(green, HIGH);  // turn the green LED on (HIGH is the voltage level)
          delay(1000);                      // wait for 1000 milliseconds
          digitalWrite(green, LOW);   // turn the green LED off by making the voltage LOW
          delay(1000);                    // wait for 1000 milliseconds

          // read the state of the pushbutton value
          buttonState = digitalRead(buttonPin);

          // check if the pushbutton is pressed. If it is, the buttonState is HIGH
          if (buttonState == HIGH) {
            digitalWrite(red, HIGH);  // turn the red LED on (HIGH is the voltage level)
            delay(1000);                      // wait for 1000 milliseconds
            digitalWrite(red, LOW);   // turn the red LED off by making the voltage LOW
            delay(1000);    // wait for 1000 milliseconds
          } else {
            for(int i = 0; i < 256; i++){ // slowly increase voltage  
              analogWrite(red, i); // change red LED voltage
              delay(9); // wait for 9 milliseconds 
            }
            digitalWrite(red, LOW);   // turn the red LED off by making the voltage LOW
            delay(1000);    // wait for 1000 milliseconds
          }
        }
